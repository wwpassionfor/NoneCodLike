### 核心组件实现

#### 点赞按钮组件 ([ThumbButton.vue](thumb-front/src/components/ThumbButton.vue))

```vue
<template>
  <button
    @click="handleToggle"
    :class="buttonClasses"
    :disabled="loading"
    class="transition-all duration-200 hover:scale-105 active:scale-95"
  >
    <!-- 点赞图标 -->
    <svg v-if="isThumbed" class="w-5 h-5 text-red-500 fill-current" viewBox="...">
      <path d="M..."/>  <!-- 红色实心爱心 -->
    </svg>
    <svg v-else class="w-5 h-5 text-gray-400" viewBox="...">
      <path d="M..."/>  <!-- 灰色空心爱心 -->
    </svg>
    
    <!-- 点赞数 -->
    <span :class="isThumbed ? 'text-red-500 font-semibold' : 'text-gray-600'" class="ml-1">
      {{ displayCount }}
    </span>
    
    <!-- 加载动画 -->
    <svg v-if="loading" class="animate-spin ml-1 w-4 h-4" viewBox="...">
      <circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4"></circle>
      <path class="opacity-75" fill="currentColor" d="..."></path>
    </svg>
  </button>
</template>

<script setup>
import { ref, computed } from 'vue';
import { useThumbStore } from '../stores/thumb';
import { useToast } from 'vue-toastification';

const props = defineProps({
  blogId: { type: Number, required: true },
  initialCount: { type: Number, default: 0 },
  initialIsThumbed: { type: Boolean, default: false }
});

const emit = defineEmits(['update', 'count-change']);

const thumbStore = useThumbStore();
const toast = useToast();
const loading = ref(false);
const isThumbed = ref(props.initialIsThumbed);
const count = ref(props.initialCount);

const buttonClasses = computed(() => [
  'inline-flex items-center px-4 py-2 rounded-full border',
  isThumbed.value 
    ? 'border-red-200 bg-red-50 hover:bg-red-100' 
    : 'border-gray-200 bg-white hover:bg-gray-50',
  loading.value && 'cursor-not-allowed opacity-60'
]);

const displayCount = computed(() => formatCount(count.value));

async function handleToggle() {
  if (loading.value || !thumbStore.isLoggedIn) {
    toast.warning('请先登录');
    return;
  }
  
  loading.value = true;
  
  try {
    if (isThumbed.value) {
      await thumbStore.undoThumb(props.blogId);
      count.value = Math.max(0, count.value - 1);
      toast.success('已取消点赞');
    } else {
      await thumbStore.doThumb(props.blogId);
      count.value += 1;
      toast.success('点赞成功 ❤️');
    }
    
    isThumbed.value = !isThumbed.value;
    emit('update', isThumbed.value);
    emit('count-change', count.value);
    
  } catch (error) {
    const msg = error.response?.data?.message || error.message;
    toast.error(msg || '操作失败，请重试');
    
    // 乐观 UI 回滚
    if (error.response?.data?.message === '用户已点赞') {
      isThumbed.value = true;  // 回滚为已点赞状态
    }
  } finally {
    loading.value = false;
  }
}

function formatCount(num) {
  if (num >= 10000) return (num / 10000).toFixed(1) + 'w';
  if (num >= 1000) return (num / 1000).toFixed(1) + 'k';
  return num.toString();
}
</script>
```

**设计亮点**：

1. **乐观 UI 更新**：点击后立即反馈，不等服务器响应
2. **错误回滚**：服务端返回错误时恢复原状态
3. **防抖动**：`loading` 状态防止重复点击
4. **数字格式化**：1234 → 1.2k，12345 → 1.2w
5. **动画过渡**：hover/active 缩放动画增强交互感

#### API 封装 ([request.js](thumb-front/src/api/request.js))

```javascript
import axios from 'axios';
import { useUserStore } from '../stores/user';

const request = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || '/api',
  timeout: 10000,
  withCredentials: true,
});

// 请求拦截器：自动添加 Token
request.interceptors.request.use(config => {
  const userStore = useUserStore();
  if (userStore.token) {
    config.headers.Authorization = `Bearer ${userStore.token}`;
  }
  return config;
}, error => Promise.reject(error));

// 响应拦截器：统一错误处理
request.interceptors.response.use(
  response => response.data,
  error => {
    if (error.response) {
      switch (error.response.status) {
        case 401:
          window.location.href = '/login';  // 未登录跳转
          break;
        case 429:
          console.error('请求过于频繁');  // 限流提示
          break;
        default:
          console.error(error.response.data?.message || '请求失败');
      }
    }
    return Promise.reject(error);
  }
);

export default request;
```

#### Pinia 状态管理 ([stores/thumb.js](thumb-front/src/stores/thumb.js))

```javascript
import { defineStore } from 'pinia';
import { thumbApi } from '../api/services';

export const useThumbStore = defineStore('thumb', {
  state: () => ({
    // 当前用户的点赞状态缓存 Map<blogId, boolean>
    thumbStatusMap: new Map(),
    isLoading: false
  }),
  
  getters: {
    isLoggedIn: () => !!localStorage.getItem('token'),
    
    isBlogThumbed: (state) => (blogId) => state.thumbStatusMap.get(blogId) ?? false
  },
  
  actions: {
    async doThumb(blogId) {
      await thumbApi.doThumb(blogId);
      this.thumbStatusMap.set(blogId, true);  // 更新本地缓存
    },
    
    async undoThumb(blogId) {
      await thumbApi.undoThumb(blogId);
      this.thumbStatusMap.set(blogId, false);
    },
    
    async batchCheckStatus(blogIds) {
      const result = await thumbApi.batchCheckStatus(blogIds);
      Object.entries(result).forEach(([blogId, status]) => {
        this.thumbStatusMap.set(Number(blogId), status);
      });
    }
  }
});
```

#### 第一轮：基线测试（无优化）


**优化措施及效果**：

| 优化项 | 措施 | QPS 提升 |
|--------|------|----------|
| 虚拟线程 | 启用 Java 21 虚拟线程 | 1200→2500 (**2.1x**) |
| 连接池调优 | HikariCP 最大连接数改为 50 | 2500→3200 (**1.3x**) |
| Redis 缓存 | 引入 Redis Hash 缓存 | 3200→8000 (**2.5x**) |

### 设计逻辑

**压测方法论**：

4. **持续迭代**：优化是一个持续的过程

## 17. 容灾演练方案

### 演练目标

验证系统在各种故障场景下的表现，确保降级策略有效。

### 演练场景矩阵

| 编号 | 故障类型 | 触发方式 | 预期行为 | RTO 目标 |
|------|----------|----------|----------|----------|
| S01 | Redis 主节点宕机 | `docker stop redis-master` | 自动切换从节点 + L2→L3 降级 | <30s |
| S02 | Redis 集群不可用 | `docker stop redis-*` | 纯 DB 模式运行 | <1min |
| S03 | Pulsar Broker 故障 | 杀掉进程 | 同步写入模式 | <1min |
| S04 | TiDB 节点宕机 | 停止 TiKV 节点 | 自动 Rebalance | <5min |
| S05 | DB 连接池耗尽 | 模拟慢 SQL | 限流 + 降级 | <30s |
| S06 | 全站流量突增 10x | JMeter 压测 | 三级降级依次触发 | 自动适应 |
| S07 | 网络分区 | iptables DROP | 本地缓存兜底 | 持续可用 |

```yaml
        condition: service_healthy

      - thumb-network

    networks:
      - thumb-network

      test: ["CMD", "mysqladmin", "ping", "-h", "localhost"]
```

### 设计逻辑

**为什么选择 Docker Compose 而非 Kubernetes？**

1. **学习成本低**：YAML 配置简单直观，团队快速上手
2. **适合中小规模**：单机或几台服务器足够运行完整集群
3. **开发友好**：本地开发环境与生产环境一致
4. **可演进性**：后续可平滑迁移至 K8s（Docker Compose 是 K8s 的子集概念）

**如果未来规模扩大，如何迁移到 K8s？**

- 将 docker-compose.yml 转换为 Kubernetes Deployment/Service YAML
- 使用 Helm Chart 管理多环境配置
- 引入 Ingress 替代 Nginx 反向代理
- 使用 Prometheus Operator 管理监控
